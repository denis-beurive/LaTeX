% This example creates commands that allow to switch on or off the boxes frames.

\documentclass{basic}
\usepackage{showframe}

% The line below is important.
% By default, the beginnings of paragraphs are indented.
\setlength{\parindent}{0pt}

% The line below sets the length of the space between the \framebox edges and its content.
% \framebox boxes are used to draw edged around the \parbox boxes.
\setlength{\fboxsep}{0px}

% This package allows simple TF THEN ELSE constructions.
\usepackage{ifthen}

% Set the value to "yes" if you want to see the borders of the boxes you use.
\newcommand{\enframe}{yes}

% Create a box that contains only one line.
% Depending on the value of \enframe, the box is framed or not.
% Param 1: width of the box.
% Param 2: justification (l|c|r}
% Param 3: the content of the box.
\newcommand{\linebox}[3]{%
    \ifthenelse{\equal{\enframe}{yes}}{%
        \framebox[#1][#2]{#3}%
    }{%
        \makebox[#1][#2]{#3}%
    }%
}%

% Create a box that contains one or more lines.
% Depending on the value of \enframe, the box is framed or not.
% Param 1: width of the box.
% Param 2: the content of the box.
\newcommand{\parabox}[2]{%
    \ifthenelse{\equal{\enframe}{yes}}{%
        \framebox[#1][c]{%
            \parbox{#1}{#2}%
        }%
    }{%
        \parbox{#1}{#2}%
    }%
}%

\begin{document}

    \renewcommand{\enframe}{no}

    \linebox{10cm}{c}{this is a linebox}

    \parabox{10cm}{this is a parabox}

    \renewcommand{\enframe}{yes}

    \linebox{10cm}{c}{this is a linebox}

    \parabox{10cm}{this is a parabox}

\end{document}
